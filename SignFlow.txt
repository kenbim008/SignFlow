      { id: 't1', name: 'NDA Agreement', fields: [{ leftPercent: 35, topPercent: 55 }, { leftPercent: 65, topPercent: 55 }], signerCount: 2 },
      { id: 't2', name: 'Sales Contract', fields: [{ leftPercent: 40, topPercent: 70 }, { leftPercent: 60, topPercent: 70 }], signerCount: 2 },
      { id: 't3', name: 'Service Agreement', fields: [{ leftPercent: 45, topPercent: 65 }], signerCount: 1 }
    ];

    // Helper: update UI & re-render
    function setActivePage(page, pushState = true) {
      activePage = page;

    function requireAuth(callback) {
      if (currentUser) callback();

    function handleSignup(email, password, plan = 'free') {
      if (!email) return false;
      currentUser = { email, plan: plan, docsRemaining: (plan === 'free' ? 3 : Infinity), documentsSigned: 0 };
      signatureFields = [];
      signers = [{ id: "s1", name: currentUser.email.split('@')[0], email: currentUser.email }];
      auditLog = [];
      reminderSettings = { enabled: false, daysBefore: 3, expirationDays: 7, reminderSent: false };
      addAudit(`Welcome ${currentUser.email}! Free tier: ${currentUser.docsRemaining === Infinity ? 'Unlimited' : currentUser.docsRemaining} documents remaining.`);
      setActivePage('dashboard');
      return true;
    }

    function handleLogin(email) {
      if (!email) return false;
      currentUser = { email, plan: 'free', docsRemaining: 3, documentsSigned: 0 };
      signers = [{ id: "s1", name: email.split('@')[0], email }];
      signatureFields = [];
      auditLog = [];
      reminderSettings = { enabled: false, daysBefore: 3, expirationDays: 7, reminderSent: false };
      addAudit(`Logged in as ${email}. Free tier: 3 free docs. Upgrade for unlimited.`);
      setActivePage('dashboard');
      return true;
    }

    function handleLogout() {
      if (expirationTimer) clearTimeout(expirationTimer);
      currentUser = null;
      signatureFields = [];
      signers = [];
      auditLog = [];
      addFieldMode = false;
      setActivePage('home');
    }

    function upgradeToUnlimited() {
      if (!currentUser) return;
      currentUser.plan = 'unlimited';
      currentUser.docsRemaining = Infinity;
      addAudit(`✨ UPGRADED to Unlimited plan: flat monthly rate, unlimited document signatures.`);
      alert('Thank you! You are now on the Unlimited plan. Flat monthly rate, unlimited signing.');

    function addAudit(message, type = 'info') {
      const entry = { timestamp: new Date().toLocaleString(), message, type };
      auditLog.unshift(entry);
      if (activePage === 'dashboard') renderDashboardContent();
    }

    function genId() { return Date.now() + '-' + Math.random().toString(36).substr(2, 6); }
    function escapeHtml(str) { if(!str) return ''; return str.replace(/[&<>]/g, function(m){if(m==='&') return '&amp;'; if(m==='<') return '&lt;'; if(m==='>') return '&gt;'; return m;}); }

    // NEW: Template functions
    function loadTemplate(templateId) {
      const template = templates.find(t => t.id === templateId);
      if (!template) return;
      signatureFields = template.fields.map((f, idx) => ({
        id: genId(),
        signerId: signers[idx % signers.length]?.id || signers[0]?.id,
        leftPercent: f.leftPercent,
        topPercent: f.topPercent,
        signedData: null,
        status: 'pending'
      }));
      addAudit(`📋 Loaded template: ${template.name} with ${signatureFields.length} signature fields.`);
      renderDashboardContent();
    }
    
    // NEW: Reminder & Expiration logic
    function setRemindersAndExpiration(enabled, daysBefore, expirationDays) {
      reminderSettings = { enabled, daysBefore, expirationDays, reminderSent: false };
      if (expirationTimer) clearTimeout(expirationTimer);
      if (enabled && expirationDays > 0) {
        expirationTimer = setTimeout(() => {
          if (signatureFields.some(f => !f.signedData)) {
            addAudit(`⏰ DOCUMENT EXPIRED: ${expirationDays} days passed without full signing.`, "warn");
            alert(`Document has expired after ${expirationDays} days. Please create a new document.`);
          }
        }, expirationDays * 24 * 60 * 60 * 1000);
        addAudit(`⏲️ Expiration set: ${expirationDays} days. Reminders: ${enabled ? `every ${daysBefore} days before expiry` : 'off'}`);
      }
      // Simulate reminder check
      if (enabled && daysBefore > 0) {
        const reminderInterval = setInterval(() => {
          if (!currentUser) { clearInterval(reminderInterval); return; }
          if (!reminderSettings.reminderSent && signatureFields.some(f => !f.signedData)) {
            addAudit(`📧 REMINDER: ${daysBefore}-day reminder sent to pending signers.`, "info");
            reminderSettings.reminderSent = true;
          }
        }, 10000); // Simulate reminder every 10 seconds for demo (in real: based on days)
        setTimeout(() => clearInterval(reminderInterval), 60000);
      }
      renderDashboardContent();
    }
    
    // NEW: Bulk send function
    function openBulkSendModal() {
      const emails = prompt("Enter comma-separated email addresses for bulk send:\ne.g., signer1@example.com, signer2@example.com");
      if (!emails) return;
      const emailList = emails.split(',').map(e => e.trim()).filter(e => e);
      bulkQueue = emailList.map(email => ({ email, name: email.split('@')[0], status: 'pending' }));
      addAudit(`📨 BULK SEND: Initiated to ${bulkQueue.length} recipients.`);
      alert(`Bulk send initiated to ${bulkQueue.length} recipients. Each will receive a signing request.`);
      // Simulate sending to first recipient
      if (bulkQueue.length > 0) {
        signers = bulkQueue.map((r, idx) => ({ id: genId(), name: r.name, email: r.email }));
        addAudit(`✅ Bulk recipients added as signers.`);
        renderDashboardContent();
      }
    }

    // Document signing core logic
    function addSignatureFieldAt(xPercent, yPercent) {
      if (!addFieldMode) return false;
      if (!signers.length) return false;
      const defaultSigner = signers[0]?.id;
      if (!defaultSigner) return false;
      const newField = { id: genId(), signerId: defaultSigner, leftPercent: Math.min(90, Math.max(5, xPercent)), topPercent: Math.min(90, Math.max(5, yPercent)), signedData: null, status: 'pending' };
      signatureFields.push(newField);
      addAudit(`➕ Added signature field for ${signers.find(s=>s.id===defaultSigner)?.name}`);
      renderDashboardContent();
      return true;
    }

    function renderFieldsOnDocument(containerElement) {
      if (!containerElement) return;
      containerElement.innerHTML = '';
      signatureFields.forEach(field => {
        const signerObj = signers.find(s => s.id === field.signerId) || signers[0];
        const isSigned = !!field.signedData;
        const fieldDiv = document.createElement('div');
        fieldDiv.className = `signature-field ${isSigned ? 'signed' : ''} absolute pointer-events-auto flex flex-col items-center justify-center`;
        fieldDiv.style.left = `${field.leftPercent}%`;
        fieldDiv.style.top = `${field.topPercent}%`;
        fieldDiv.style.transform = 'translate(-50%, -50%)';
        fieldDiv.style.minWidth = '130px';
        let contentHtml = '';
        if (isSigned) {
          if (field.signedData.type === 'typed') contentHtml = `<div class="text-xs text-green-700 font-semibold">✓ Signed</div><div class="text-sm" style="font-family: 'Brush Script MT', cursive;">${field.signedData.html}</div>`;
          else if (field.signedData.type === 'draw') contentHtml = `<div class="text-xs text-green-600">✍️ Signature</div><img src="${field.signedData.imgData}" style="max-height: 40px; max-width: 110px;" />`;
        } else contentHtml = `<div class="text-xs font-semibold text-gray-500">🔏 Pending</div><div class="text-[11px]">${escapeHtml(signerObj.name)}</div>`;
        fieldDiv.innerHTML = `<div class="flex justify-between w-full items-center gap-1 text-[10px] mb-1"><span class="bg-gray-200 px-1 rounded">${escapeHtml(signerObj.name.split(' ')[0])}</span><div class="flex gap-1">${!isSigned ? `<button class="sign-field-btn text-blue-600 hover:bg-blue-100 rounded-full w-5 h-5 flex items-center justify-center" data-id="${field.id}"><i class="fas fa-pen"></i></button>` : ''}<button class="delete-field-btn text-red-400 hover:text-red-700 rounded-full w-5 h-5 flex items-center justify-center" data-id="${field.id}"><i class="fas fa-times"></i></button></div></div><div class="text-center w-full">${contentHtml}</div>`;
        containerElement.appendChild(fieldDiv);
      });
      document.querySelectorAll('.sign-field-btn').forEach(btn => btn.addEventListener('click', (e) => { e.stopPropagation(); const id = btn.getAttribute('data-id'); openSignatureModal(id); }));
      document.querySelectorAll('.delete-field-btn').forEach(btn => btn.addEventListener('click', (e) => { e.stopPropagation(); const id = btn.getAttribute('data-id'); deleteFieldById(id); }));
    }

    function deleteFieldById(id) { signatureFields = signatureFields.filter(f => f.id !== id); addAudit(`Removed signature field`); renderDashboardContent(); }

    function openSignatureModal(fieldId) {
      const field = signatureFields.find(f => f.id === fieldId);
      if (!field || field.signedData) return;
      currentFieldIdForModal = fieldId;
      const signer = signers.find(s => s.id === field.signerId) || signers[0];
      document.getElementById('modalSignerName').innerText = signer.name;
      document.getElementById('typedSignatureInput').value = signer.name;
      if (canvasCtx && canvasEl) { canvasCtx.clearRect(0, 0, canvasEl.width, canvasEl.height); canvasCtx.fillStyle = "white"; canvasCtx.fillRect(0, 0, canvasEl.width, canvasEl.height); currentDrawDataUrl = null; }
      document.getElementById('signatureModal').classList.remove('hidden');
    }

    function signFieldWithTyped() {
      if (!currentFieldIdForModal) return;
      const field = signatureFields.find(f => f.id === currentFieldIdForModal);
      if (!field) return;
      const typedValue = document.getElementById('typedSignatureInput').value.trim();
      if (!typedValue) return alert("Enter signature");
      field.signedData = { type: 'typed', html: typedValue };
      addAudit(`✍️ Signed by ${signers.find(s=>s.id===field.signerId)?.name}`);
      closeModal();
      renderDashboardContent();
      checkDocumentCompletion();
    }

    function signFieldWithDraw() {
      if (!currentFieldIdForModal) return;
      const field = signatureFields.find(f => f.id === currentFieldIdForModal);
      if (!field || !currentDrawDataUrl) return alert("Draw signature first");
      field.signedData = { type: 'draw', imgData: currentDrawDataUrl };
      addAudit(`🖌️ Drew signature by ${signers.find(s=>s.id===field.signerId)?.name}`);
      closeModal();
      renderDashboardContent();
      checkDocumentCompletion();
    }

    function checkDocumentCompletion() {
      if (!currentUser) return;
      const allSigned = signatureFields.length > 0 && signatureFields.every(f => f.signedData);
      if (allSigned && signatureFields.length) {
        if (currentUser.plan === 'free' && currentUser.docsRemaining > 0 && currentUser.docsRemaining !== Infinity) {
          currentUser.docsRemaining -= 1;
          currentUser.documentsSigned = (currentUser.documentsSigned || 0) + 1;
          addAudit(`📄 Document fully signed! Free tier: ${currentUser.docsRemaining} document(s) left.`);
          if (currentUser.docsRemaining === 0) alert("You've used all 3 free documents! Please upgrade to unlimited plan for continued signing.");
          else alert("Document completed! Audit trail saved.");
        } else if (currentUser.plan === 'unlimited') {
          addAudit(`📄 Document fully signed (Unlimited plan).`);
          alert("Document fully executed! Unlimited plan active.");
        } else if (currentUser.plan === 'free' && currentUser.docsRemaining <= 0) {
          addAudit(`⚠️ No free documents left. Please upgrade to continue.`, "warn");
          alert("You have used your 3 free documents. Upgrade to unlimited plan to keep signing.");
        } else { addAudit(`Document signed.`); alert("Document complete."); }
      }
    }

    function closeModal() { document.getElementById('signatureModal')?.classList.add('hidden'); currentFieldIdForModal = null; }

    function initCanvasDrawing() {
      canvasEl = document.getElementById('signCanvas');
      if (!canvasEl) return;
      canvasCtx = canvasEl.getContext('2d');
      function resizeAndClear() { canvasEl.width = canvasEl.parentElement.clientWidth - 20; canvasEl.height = 150; canvasCtx.fillStyle = "white"; canvasCtx.fillRect(0, 0, canvasEl.width, canvasEl.height); canvasCtx.strokeStyle = "#333"; canvasCtx.lineWidth = 2; canvasCtx.lineCap = "round"; }
      resizeAndClear(); window.addEventListener('resize', () => resizeAndClear());
      let drawingFlag = false;
      function getCoords(e) { const rect = canvasEl.getBoundingClientRect(); const scaleX = canvasEl.width / rect.width; const scaleY = canvasEl.height / rect.height; let clientX, clientY; if (e.touches) { clientX = e.touches[0].clientX; clientY = e.touches[0].clientY; } else { clientX = e.clientX; clientY = e.clientY; } let x = (clientX - rect.left) * scaleX; let y = (clientY - rect.top) * scaleY; return { x: Math.min(Math.max(0, x), canvasEl.width), y: Math.min(Math.max(0, y), canvasEl.height) }; }
      function start(e) { drawingFlag = true; const pos = getCoords(e); canvasCtx.beginPath(); canvasCtx.moveTo(pos.x, pos.y); canvasCtx.lineTo(pos.x, pos.y); canvasCtx.stroke(); e.preventDefault(); }
      function draw(e) { if (!drawingFlag) return; const pos = getCoords(e); canvasCtx.lineTo(pos.x, pos.y); canvasCtx.stroke(); canvasCtx.beginPath(); canvasCtx.moveTo(pos.x, pos.y); e.preventDefault(); }
      function stop() { drawingFlag = false; canvasCtx.beginPath(); currentDrawDataUrl = canvasEl.toDataURL(); }
      canvasEl.addEventListener('mousedown', start); canvasEl.addEventListener('mousemove', draw); canvasEl.addEventListener('mouseup', stop); canvasEl.addEventListener('touchstart', start); canvasEl.addEventListener('touchmove', draw); canvasEl.addEventListener('touchend', stop);
      document.getElementById('clearCanvasBtn')?.addEventListener('click', () => { canvasCtx.clearRect(0, 0, canvasEl.width, canvasEl.height); canvasCtx.fillStyle = "white"; canvasCtx.fillRect(0, 0, canvasEl.width, canvasEl.height); currentDrawDataUrl = null; });
    }

'♾️ Unlimited' : `📄 ${currentUser.docsRemaining} left`}</span><button id="logoutBtn" class="text-sm bg-gray-200 px-3 py-1 rounded-full hover:bg-gray-300"><i class="fas fa-sign-out-alt"></i> Logout</button><button data-nav="dashboard" class="bg-blue-600 text-white px-4 py-1.5 rounded-full text-sm shadow">📄 Dashboard</button>` : `<button data-nav="login" class="text-gray-700 text-sm font-medium">Sign In</button><button data-nav="signup" class="bg-blue-600 text-white px-4 py-1.5 rounded-full text-sm shadow">Start Free</button>`}</div></div></header>`;
